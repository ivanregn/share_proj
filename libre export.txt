procedure ExportToLibreOfficeLinux(ADataSource: TDataSource; const ATemplatePath, ASavePath: string);
var
  Zip: TZipFile;
  XmlContent: TStringList;
  DataSet: TDataSet;
  ColIndex: Integer;
  ContentStr: string;
  InsertPos: Integer;
  FieldValue: string;
  NumValue: string;
  XmlStream: TStream;
  NewXmlStream: TStringStream;
  Header: TZipHeader;
  DataRowsBuilder: TStringBuilder;
  RowsFound: Integer;
  TargetDir: string;
begin
  // 1. Проверки входных данных
  if (ADataSource = nil) or (ADataSource.DataSet = nil) then
    raise Exception.Create('DataSource или DataSet не инициализированы.');

  DataSet := ADataSource.DataSet;
  if not DataSet.Active then
    DataSet.Open;

  // Проверяем и создаем папку назначения (например, .../output/)
  TargetDir := TPath.GetDirectoryName(ASavePath);
  if (TargetDir <> '') and (not TDirectory.Exists(TargetDir)) then
    TDirectory.CreateDirectory(TargetDir);

  // Если старый файл отчета уже существует — удаляем его
  if TFile.Exists(ASavePath) then
    TFile.Delete(ASavePath);

  // ШАГ-ФИКС: Вместо ручной сборки ZIP с нуля, мы делаем точную копию оригинального шаблона.
  // Это гарантирует, что файл "mimetype" и структуры папок останутся нетронутыми и валидными!
  TFile.Copy(ATemplatePath, ASavePath);

  Zip := TZipFile.Create;
  XmlContent := TStringList.Create;
  DataRowsBuilder := TStringBuilder.Create;
  try
    // 2. Открываем скопированный файл в режиме добавления/изменения (zmReadWrite)
    Zip.Open(ASavePath, zmReadWrite);
    
    // Читаем исходный XML-контент в память
    Zip.Read('content.xml', XmlStream, Header);
    try
      XmlContent.LoadFromStream(XmlStream, TEncoding.UTF8);
    finally
      XmlStream.Free;
    end;

    ContentStr := XmlContent.Text;

    // 3. Подготовка XML-строк данных из DataSet
    try
      DataSet.DisableControls;
      try
        DataSet.First;
        while not DataSet.Eof do
        begin
          DataRowsBuilder.Append('<table:table-row>');
          for ColIndex := 0 to DataSet.FieldCount - 1 do
          begin
            FieldValue := DataSet.Fields[ColIndex].AsString;
            // Экранируем спецсимволы XML для безопасности данных
            FieldValue := FieldValue.Replace('&', '&amp;').Replace('<', '&lt;').Replace('>', '&gt;');

            // Проверяем тип данных
            if DataSet.Fields[ColIndex].DataType in [ftInteger, ftFloat, ftCurrency, ftLargeint, ftBCD] then
            begin
              NumValue := DataSet.Fields[ColIndex].AsFloat.ToString(TFormatSettings.Invariant);
              DataRowsBuilder.Append('<table:table-cell office:value-type="float" office:value="' + 
                NumValue + '"><text:p>' + FieldValue + '</text:p></table:table-cell>');
            end
            else
            begin
              DataRowsBuilder.Append('<table:table-cell office:value-type="string"><text:p>' + 
                FieldValue + '</text:p></table:table-cell>');
            end;
          end;
          DataRowsBuilder.Append('</table:table-row>');
          DataSet.Next;
        end;
      finally
        DataSet.EnableControls;
      end;

      // 4. Поиск места вставки (Отсчитываем ровно 2 строки существующей шапки шаблона)
      InsertPos := 0;
      RowsFound := 0;
      while RowsFound < 3 do
      begin
        InsertPos := ContentStr.IndexOf('</table:table-row>', InsertPos);
        if InsertPos = -1 then Break;
        InsertPos := InsertPos + Length('</table:table-row>');
        Inc(RowsFound);
      end;

      // Безопасный откат к началу таблицы, если строк в шаблоне не нашлось
      if InsertPos = -1 then
      begin
        InsertPos := ContentStr.IndexOf('<table:table');
        if InsertPos <> -1 then
          InsertPos := ContentStr.IndexOf('>', InsertPos) + 1;
      end;

      if InsertPos = -1 then
        raise Exception.Create('Ошибка структуры шаблона: в файле отсутствует табличная структура.');

      // Вставляем подготовленные данные строго со строки 3
      ContentStr := ContentStr.Insert(InsertPos, DataRowsBuilder.ToString);
    end;
    finally
      DataRowsBuilder.Free;
    end;

    // 5. Перезаписываем ИСКЛЮЧИТЕЛЬНО файл content.xml внутри архива.
    // Метод Add в режиме zmReadWrite автоматически обновляет/заменяет существующий файл,
    // не затрагивая порядок других файлов и структуру сжатия mimetype!
    XmlContent.Text := ContentStr;
    NewXmlStream := TStringStream.Create(XmlContent.Text, TEncoding.UTF8);
    try
      Zip.Add(NewXmlStream, 'content.xml');
    finally
      NewXmlStream.Free;
    end;

    Zip.Close;
  finally
    XmlContent.Free;
    Zip.Free;
  end;
end;

procedure TForm1.ButtonExportClick(Sender: TObject);
var
  AppDir: string;
  TemplateFile: string;
  OutputFile: string;
begin
  // Получаем текущую папку запущенного приложения
  AppDir := ExtractFilePath(ParamStr(0));

  // Путь к файлу шаблона (должен лежать рядом с вашим EXE)
  TemplateFile := TPath.Combine(AppDir, 'template_test.ods');
  
  // Безопасно формируем путь к подпапке output (которая создастся сама)
  OutputFile := TPath.Combine(AppDir, TPath.Combine('output', 'template_test.ods'));

  try
    ExportToLibreOfficeLinux(DataSource1, TemplateFile, OutputFile);
    ShowMessage('Экспорт успешно завершен! Файл сохранен в папку output.');
  except
    on E: Exception do
      ShowMessage('Произошла ошибка при экспорте: ' + E.Message);
  end;
end;
